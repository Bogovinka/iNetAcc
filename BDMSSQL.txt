	

CREATE TABLE Logins(
	ID int primary key IDENTITY(1,1),
	Surname nvarchar(250),
	Name nvarchar(250),
	LastName nvarchar(250),
	Phone nvarchar(250),
	Login nvarchar(250),
	Password nvarchar(250),
	Permission int);

CREATE TABLE Model(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));
	
CREATE TABLE Brand(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));

CREATE TABLE Conditions(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));

CREATE TABLE Type_Device(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));

CREATE TABLE Equipment(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250),
	IP nvarchar(250),
	MAC nvarchar(250),
	Type_Device_id int,
	Conditions_id int,
	Address nvarchar(250),
	Note nvarchar(250),
	ID_in_item int,
	Login nvarchar(250),
	Password nvarchar(250),
	SNMP nvarchar(250),
	Serial_num nvarchar(250),
	Num_vlan nvarchar(250),
	Brand_id int,
	Model_id int,
	FOREIGN KEY (Type_Device_id) REFERENCES Type_Device(ID),
    	FOREIGN KEY (Conditions_id) REFERENCES Conditions(ID),
    	FOREIGN KEY (Brand_id) REFERENCES Brand(ID),
    	FOREIGN KEY (Model_id) REFERENCES Model(ID));



CREATE TABLE Warehouse(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));

CREATE TABLE EquipmentWarehouseType(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));

CREATE TABLE EquipmentWarehouseStatus(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));

CREATE TABLE EquipmentWarehouse(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250),
	EquipmentWarehouseTypeID int FOREIGN KEY REFERENCES EquipmentWarehouseType(ID),
	SerialNum nvarchar(250),
	ConditionsID int FOREIGN KEY REFERENCES Conditions(ID),
	Model nvarchar(250),
	StatusID int FOREIGN KEY REFERENCES EquipmentWarehouseStatus(ID));

CREATE TABLE Client(
	ID int primary key IDENTITY(1,1),
	Surname nvarchar(250),
	Name nvarchar(250),
	LastName nvarchar(250),
	Phone nvarchar(250));	
	
CREATE TABLE ClientContracts(
	ID int primary key IDENTITY(1,1),
	ClientID int FOREIGN KEY REFERENCES Client(ID),
	Contract nvarchar(250));

CREATE TABLE EquipmentWarehouseClient(
	ID int primary key IDENTITY(1,1),
	EquipmentWarehouseID int FOREIGN KEY REFERENCES EquipmentWarehouse(ID),
	ClientID int FOREIGN KEY REFERENCES Client(ID),
	Size int);

CREATE TABLE EquipmentWarehouseMaster(
	ID int primary key IDENTITY(1,1),
	EquipmentWarehouseID int FOREIGN KEY REFERENCES EquipmentWarehouse(ID),
	MasterID int FOREIGN KEY REFERENCES Logins(ID),
	Size int);	
	
CREATE TABLE EquipmentWarehouseS(
	ID int primary key IDENTITY(1,1),
	EquipmentWarehouseID int FOREIGN KEY REFERENCES EquipmentWarehouse(ID),
	WarehouseID int FOREIGN KEY REFERENCES Warehouse(ID),
	Size int);

CREATE TABLE Unit(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));
	
CREATE TABLE TMC(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250),
	Size int,
	UnitID int FOREIGN KEY REFERENCES Unit(ID),
	WarehouseID int FOREIGN KEY REFERENCES Warehouse(ID));

CREATE TABLE TypeTask(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));

CREATE TABLE StatusTask(
	ID int primary key IDENTITY(1,1),
	Name nvarchar(250));

CREATE TABLE Task(
	ID int primary key IDENTITY(1,1),
	IDofType int not null,
	Contract int FOREIGN KEY REFERENCES ClientContracts(ID),
	DateStart date,
	DateEnt date,
	Creator nvarchar(250),
	MasterID int FOREIGN KEY REFERENCES Logins(ID),
	StatusTaskID int FOREIGN KEY REFERENCES StatusTask(ID),
	Addres nvarchar(250),
	TypeID int FOREIGN KEY REFERENCES TypeTask(ID),
	PortID int FOREIGN KEY REFERENCES Equipment(ID),
	Note nvarchar(max));
	
CREATE TABLE TaskTMC(
	ID int primary key IDENTITY(1,1),
	TaskID int FOREIGN KEY REFERENCES Task(ID),
	TMCID int FOREIGN KEY REFERENCES TMC(ID),
	Size int);
	
CREATE TABLE TaskEquipment(
	ID int primary key IDENTITY(1,1),
	TaskID int FOREIGN KEY REFERENCES Task(ID),
	EquipmentWarehouse int FOREIGN KEY REFERENCES EquipmentWarehouse(ID));
